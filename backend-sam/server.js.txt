// backend/server.js
const express = require('express');
const cors = require('cors');
const admin = require('firebase-admin');
const app = express();

// Path to your service account key file
// Make sure this file is in the same folder as this server.js.txt file
const serviceAccount = require("./disrupt-53691-firebase-adminsdk-fbsvc-7410fd769f.json");

// Initialize Firebase Admin SDK
admin.initializeApp({
  credential: admin.credential.cert(serviceAccount)
});

// Use middleware
app.use(cors());
app.use(express.json());

// Example API endpoint to get all users
app.get('/api/users', async (req, res) => {
  try {
    const listUsersResult = await admin.auth().listUsers();
    res.status(200).json(listUsersResult.users);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

const PORT = process.env.PORT || 5000;
app.listen(PORT, () => {
  console.log(`Server is running on port ${PORT}`);
});
